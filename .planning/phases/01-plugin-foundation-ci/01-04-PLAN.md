```typescript
import { z } from 'zod';
import fs from 'node:fs/promises';
import path from 'node:path';

// ────────────────────────────────────────────────────────────────────────────
// Marketplace + plugin schemas
// ────────────────────────────────────────────────────────────────────────────

const KebabCase = z.string().regex(/^[a-z0-9][a-z0-9-]*$/, 'must be kebab-case (lowercase letters/digits/hyphens, start with letter/digit)');

const RelativeSource = z.string().regex(/^\.\//, "source must start with './'").refine(
  (s) => !s.includes('..'),
  { message: "source must not contain '..'" },
);

const MarketplaceSchema = z.object({
  $schema: z.string().optional(),
  name: KebabCase,
  description: z.string().optional(),
  owner: z.object({
    name: z.string().min(1),
    email: z.string().email().optional(),
  }),
  plugins: z.array(z.object({
    name: KebabCase,
    source: RelativeSource,
    description: z.string().optional(),
    version: z.string().optional(),
    homepage: z.string().optional(),
    repository: z.string().optional(),
    license: z.string().optional(),
    keywords: z.array(z.string()).optional(),
    category: z.string().optional(),
  })).min(1),
});

const PluginSchema = z.object({
  name: KebabCase,
  description: z.string().optional(),
  version: z.string().optional(),
  author: z.object({ name: z.string(), email: z.string().email().optional() }).optional(),
  homepage: z.string().optional(),
  repository: z.string().optional(),
  license: z.string().optional(),
  keywords: z.array(z.string()).optional(),
});

const RESERVED_MARKETPLACE_NAMES = new Set([
  'claude-code-marketplace',
  'claude-code-plugins',
  'claude-plugins-official',
  'anthropic-marketplace',
  'anthropic-plugins',
  'agent-skills',
  'knowledge-work-plugins',
  'life-sciences',
]);

// ────────────────────────────────────────────────────────────────────────────
// SKILL.md frontmatter schema
// ────────────────────────────────────────────────────────────────────────────

const SkillFrontmatterSchema = z.object({
  name: z.string().regex(/^[a-z0-9-]+$/).max(64),
  description: z.string().min(1),
  when_to_use: z.string().min(1),
  'allowed-tools': z.string().min(1),
  'disable-model-invocation': z.boolean().optional(),
  context: z.enum(['inline', 'fork']).optional(),
  agent: z.string().optional(),
}).refine(
  (fm) => (fm.description.length + fm.when_to_use.length) <= 1536,
  { message: 'combined description + when_to_use must be ≤ 1,536 chars' },
);

// ────────────────────────────────────────────────────────────────────────────
// Minimal YAML-frontmatter parser (no gray-matter dep needed for Phase 1)
// ────────────────────────────────────────────────────────────────────────────

function parseFrontmatter(md: string): Record<string, unknown> {
  const m = md.match(/^---\n([\s\S]*?)\n---/);
  if (!m) throw new Error('SKILL.md missing YAML frontmatter (--- block)');
  const body = m[1];
  const out: Record<string, unknown> = {};
  for (const line of body.split('\n')) {
    const kv = line.match(/^([A-Za-z][A-Za-z0-9_-]*):\s*(.*)$/);
    if (!kv) continue;
    const [, key, raw] = kv;
    const v = raw.trim();
    if (v === 'true') out[key] = true;
    else if (v === 'false') out[key] = false;
    else out[key] = v;
  }
  return out;
}

// ────────────────────────────────────────────────────────────────────────────
// Main
// ────────────────────────────────────────────────────────────────────────────

const EXPECTED_SKILLS = ['init', 'contract', 'test', 'deploy', 'frontend'] as const;
const PLUGIN_DIR = 'plugins/zama-skills';
const SKILLS_ROOT = `${PLUGIN_DIR}/skills`;

async function main() {
  const errors: string[] = [];

  // 1. marketplace.json
  const mpRaw = await fs.readFile('.claude-plugin/marketplace.json', 'utf8');
  const mp = MarketplaceSchema.parse(JSON.parse(mpRaw));
  if (RESERVED_MARKETPLACE_NAMES.has(mp.name)) {
    errors.push(`marketplace.name "${mp.name}" is on the reserved list`);
  }
  if (mp.plugins[0].source !== `./${PLUGIN_DIR}`) {
    errors.push(`plugins[0].source must be "./${PLUGIN_DIR}", got "${mp.plugins[0].source}"`);
  }
  if ('version' in mp.plugins[0]) {
    errors.push('plugins[0].version should be omitted during dev (commit SHA = version); pin only at submission');
  }

  // 2. plugin.json
  const plRaw = await fs.readFile(`${PLUGIN_DIR}/.claude-plugin/plugin.json`, 'utf8');
  const pl = PluginSchema.parse(JSON.parse(plRaw));
  if (pl.name !== 'zama-skills') errors.push(`plugin.name must be "zama-skills", got "${pl.name}"`);
  if ('version' in pl) {
    errors.push('plugin.json version should be omitted during dev; pin only at submission');
  }

  // 3. .claude-plugin directory must contain ONLY plugin.json
  const pluginCpEntries = await fs.readdir(`${PLUGIN_DIR}/.claude-plugin`);
  if (pluginCpEntries.length !== 1 || pluginCpEntries[0] !== 'plugin.json') {
    errors.push(`${PLUGIN_DIR}/.claude-plugin must contain only plugin.json, got: ${pluginCpEntries.join(', ')}`);
  }

  // 4. All 5 SKILL.md files exist with correct invariants
  for (const slug of EXPECTED_SKILLS) {
    const p = path.join(SKILLS_ROOT, slug, 'SKILL.md');
    let content: string;
    try {
      content = await fs.readFile(p, 'utf8');
    } catch {
      errors.push(`Missing skill file: ${p}`);
      continue;
    }
    let fm: ReturnType<typeof SkillFrontmatterSchema.parse>;
    try {
      const parsed = parseFrontmatter(content);
      fm = SkillFrontmatterSchema.parse(parsed);
    } catch (e) {
      errors.push(`${p} frontmatter invalid: ${e instanceof Error ? e.message : String(e)}`);
      continue;
    }
    if (fm.name !== slug) errors.push(`${p}: name "${fm.name}" must match folder "${slug}"`);
    if (slug === 'deploy') {
      if (fm['disable-model-invocation'] !== true) {
        errors.push(`${p}: deploy skill MUST have disable-model-invocation: true (PLUGIN-03)`);
      }
    } else {
      if (fm['disable-model-invocation'] === true) {
        errors.push(`${p}: only deploy skill should have disable-model-invocation: true`);
      }
    }
    if (slug === 'init') {
      if (fm.context !== 'fork') {
        errors.push(`${p}: init skill MUST have context: fork (CONTEXT D-frontmatter)`);
      }
    } else {
      if (fm.context === 'fork') {
        errors.push(`${p}: only init skill should have context: fork`);
      }
    }
  }

  if (errors.length) {
    console.error('✗ Validation FAILED:\n');
    for (const e of errors) console.error('  • ' + e);
    process.exit(1);
  }
  console.log('✓ marketplace + plugin + 5 SKILL.md frontmatters valid');
}

main().catch((e) => {
  console.error('✗ Validator crashed:', e instanceof Error ? e.message : e);
  process.exit(1);
});
```

Create `.github/workflows/ci.yml`. Mandatory steps: install (`npm ci`), validate (zod), typecheck, test. Best-effort step: `claude plugin validate` (only if package name verified at runtime; never blocks the pipeline).

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:

permissions:
  contents: read

jobs:
  validate:
    name: Validate plugin + run tests
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - uses: actions/checkout@v5

      - name: Setup Node.js 20
        uses: actions/setup-node@v5
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies (lockfile-strict)
        run: npm ci --no-audit --no-fund

      - name: Validate manifests + SKILL.md frontmatter (zod)
        run: npm run validate

      - name: Type-check
        run: npx tsc --noEmit

      - name: Run tests
        run: npm test

      - name: Best-effort — official Claude plugin validate CLI
        # Phase 1 best-effort: try the official CLI but never block the build.
        # Phase 6 should re-evaluate once the CLI distribution package is confirmed.
        # Tracked as RESEARCH.md A4 / Open Question 2.
        continue-on-error: true
        run: |
          if npx --yes --package=@anthropic-ai/claude-code -- claude plugin validate . 2>/dev/null; then
            echo "✓ Official CLI validation passed"
          else
            echo "::notice::Official 'claude plugin validate' CLI not available in this runner; relying on zod validator (mandatory step above is the gate)."
          fi
```

Critical hygiene:
- `permissions: contents: read` — minimum required; no write tokens leaked.
- NO secrets referenced (`NPM_TOKEN`, `GITHUB_TOKEN` write, etc.) — Phase 6 adds publish workflow with `NPM_TOKEN`.
- `npm ci` (NOT `npm install`) — strict lockfile, fails on drift.
- `continue-on-error: true` ONLY on the best-effort CLI step. The zod validator step is the actual gate and runs first.
- Single job — no matrix. Node 20 only per RESEARCH.md Open Question 4 ("Layer B will only support 20+, so Node 22 doesn't add value for Layer A scaffolding").
